cmake_minimum_required(VERSION 3.16)
project(MyRenderer)

set(CMAKE_C_STANDARD 11)

find_package(glfw3 REQUIRED)

add_executable(MyRenderer
    src/main.c
    glad/src/glad.c
)

target_include_directories(MyRenderer PRIVATE
    glad/include
)

target_link_libraries(MyRenderer
    glfw
    "-framework OpenGL"
    "-framework Cocoa"
    "-framework IOKit"
)
